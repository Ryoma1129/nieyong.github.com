2. More -> 意见反馈，跳到发邮件界面，点击取消或完成，都已经返回到WM Assistant，而不是返回到邮件程序。这个是容易实现的，试用一下“我在这儿”这个应用，点击+号，选择邮件。

MFMailComposeViewController *mailPicker = [[MFMailComposeViewController alloc] init]; mailPicker.mailComposeDelegate = self;
//设置主题 [mailPicker setSubject: @"eMail主题"]; //添加收件人 NSArray *toRecipients = [NSArray arrayWithObject: @"app@dji-innovations.com"]; [mailPicker setToRecipients: toRecipients];
NSString *emailBody = @"<font color='red'>eMail</font> 正文"; [mailPicker setMessageBody:emailBody isHTML:YES];
UIViewController *theMailViewController = [[mailPicker viewControllers] lastObject]; [self.navigationController pushViewController:theMailViewController animated:YES];
//[self presentModalViewController: mailPicker animated:YES];
[mailPicker release];
